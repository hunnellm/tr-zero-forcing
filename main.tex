\begin{frame}{Standard Zero Forcing: Core Idea}
Let $G=(V,E)$ be a simple graph.
\begin{itemize}
  \item Each vertex is colored either black (active) or white (inactive).
  \item Start with an initial black set $B_0\subseteq V$.
  \item Apply the color-change rule repeatedly.
\end{itemize}

\vspace{0.3em}
\textbf{Color-change rule:} If a black vertex $u$ has exactly one white neighbor $v$, then $u$ \emph{forces} $v$ to become black. We write $u\to v$.
\end{frame}

\begin{frame}{Process and Terminology}
\begin{itemize}
  \item A sequence of valid forces is a \emph{forcing process}.
  \item If every vertex eventually becomes black, then $B_0$ is a \emph{zero forcing set}.
  \item The minimum size of such a set is the \emph{zero forcing number}:
\end{itemize}

\[
Z(G)=\min\{\abs{B}: B\subseteq V,\ B \text{ is a zero forcing set}\}.
\]

\vspace{0.3em}
\textbf{Goal:} choose a small initial black set that propagates to all of $V$.
\end{frame}

\column{0.44\textwidth}
\begin{itemize}
  \item With $B_0=\{r,b\}$, vertex $b$ can force $r$ or one leaf depending on current colors.
  \item If the wrong vertices are initially black, propagation can stall.
  \item Strategic initial placement is the heart of zero forcing.
\end{itemize}
\end{columns}

\begin{frame}{Takeaway and Transition}
\begin{itemize}
  \item Standard zero forcing tracks deterministic propagation via the unique-white-neighbor rule.
  \item The central optimization problem is minimizing $\abs{B_0}$, giving $Z(G)$.
  \item This classical process is our baseline for $(t,r)$ zero forcing, where force dynamics are generalized.
\end{itemize}
