\begin{frame}{Standard Zero Forcing: Core Idea}
\small
Let $G=(V,E)$ be a simple graph.
\begin{itemize}
  \item Each vertex is colored either light-blue (active) or white (inactive).
  \item Start with an initial active set $B_0\subseteq V$.
  \item Apply the color-change rule repeatedly.
\end{itemize}

\vspace{0.2em}
\textbf{Color-change rule:} If an active vertex $u$ has exactly one white neighbor $v$, then $u$ \emph{forces} $v$ to become active. We write $u\to v$.
\end{frame}

\begin{frame}{Process and Terminology}
\small
\begin{itemize}
  \item A sequence of valid forces is a \emph{forcing process}.
  \item If every vertex eventually becomes active, then $B_0$ is a \emph{zero forcing set}.
  \item The minimum size of such a set is the \emph{zero forcing number}:
\end{itemize}

\[
Z(G)=\min\{\abs{B}: B\subseteq V,\ B \text{ is a zero forcing set}\}.
\]

\vspace{0.2em}
\textbf{Goal:} choose a small initial active set that propagates to all of $V$.
\end{frame}

\begin{frame}{Why the Path Example Works}
\small
\begin{columns}[T,totalwidth=\textwidth]
\column{0.53\textwidth}
\begin{itemize}
  \item Start with one active endpoint.
  \item At each step, the current frontier vertex has exactly one white neighbor.
  \item This keeps creating the next valid force.
\end{itemize}

1\to2\to3\to4\to5

\begin{frame}{Takeaway and Transition}
\small
\begin{itemize}
  \item Standard zero forcing tracks deterministic propagation via the unique-white-neighbor rule.
  \item The central optimization problem is minimizing $\abs{B_0}$, giving $Z(G)$.
  \item This classical process is our baseline for $(t,r)$ zero forcing, where force dynamics are generalized.
\end{itemize}
